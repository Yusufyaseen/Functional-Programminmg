
AddOne(a) {
  return a+1;
}
Square(a) {
  return a*a;
} 


SubstractTen(a) {
  return a-10;
}

Function compose(f1,f2,f3)
{
  return (x) 
  {
    return f3(f2(f1(x)));
    
  };
}
Function MyComposedFunction = compose(AddOne,Square,SubstractTen);
void main() {
  List b = [3,5,7,8];
 
  var c = b.map(MyComposedFunction).toList();
  print(c);
}
